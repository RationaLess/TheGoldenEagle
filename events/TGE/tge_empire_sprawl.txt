namespace = empire_sprawl

empire_sprawl.1 = {
	type = country_event
	placement = scope:subject_scope.capital
	
	title = empire_sprawl.1.t
	desc = empire_sprawl.1.d
	flavor = empire_sprawl.1.f

	cooldown = { months = short_modifier_time }
	
	event_image = {
		texture = "gfx/event_pictures/diplomacy_meso.dds"
	}
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"

	icon = "gfx/interface/icons/event_icons/event_default.dds"
	
	duration = 3
	
	trigger = {
		empire_sprawl_deficit > 0.2
		any_country = {
			is_subject_type = subject_type_tributary
			is_subject_of = root
		}
	}

	immediate = {
		random_country = {
			limit = {
			is_subject_type = subject_type_tributary
			is_subject_of = root
			}
		save_scope_as = subject_scope
		}
	}

	option = {
		name = empire_sprawl.1.a
		
		create_incident	 = {
			country = scope:subject_scope
			value = 2
			}
		
		change_relations = {
			country = scope:subject_scope
			value = -30
			}
		
		ruler = {
			add_modifier = {
			name = tyranny
			months = 120
			}
		}
		
		scope:subject_scope = {
			kill_population_percent = 0.01
		}
	}
	option = {
		name = empire_sprawl.1.b
		default_option = yes
		
		
		ruler = {
			add_modifier = {
			name = kindness
			months = 120
			}
		}
		
		add_modifier = {
			name = forgave_loss_of_tribute
			months = 60
		}
	}
}
