# This is executed last among all history
GLOBAL = {
	
	every_country = {
		onaction_empire_sprawl_calc = yes
		
		
		#Civilization Level JEs
		
		set_variable = {
			name = civilization_value
			value = 0
			}
		set_variable = {
			name = civilization_level
			value = 0
			}
		set_variable = {
			name = confederations_num
			value = 0
			}
	
	
		IF = {
			limit = {
				is_ai = yes 
			}
			IF = {
				limit = {
					has_game_rule = easy_difficulty
				}
			add_modifier = axolotl_difficulty
			}
			IF = {
				limit = {
					has_game_rule = hard_difficulty
				}
			add_modifier = ocelotl_difficulty
			}
			IF = {
				limit = {
					has_game_rule = very_hard_difficulty
				}
			add_modifier = coatl_difficulty
			}
		
		}
	
		#Freemen from
		IF = {
			limit = {
				any_primary_culture = {
					has_discrimination_trait = settled_tribe
					}	
			}
			
			ig:ig_freemen = {
				remove_ideology = ideology_sedentary_legacy
				add_ideology = ideology_migratory_legacy
				}
			
		}
		
		#Giving the Right Civ Level JE
		IF = { #Settlement
			limit = {
				has_law = law_type:law_tribe
			}
			add_journal_entry = { type = je_settlement_civ_level  }
		}
		ELSE = { #Urban
			add_journal_entry = { type = je_urban_civ_level  }
		}
		
	
		#Every Country should start with at least some loyalists
		add_loyalists = {
			value = 0.5
			}
	
	}
	
	every_state = {
		add_modifier = {
			name = state_minor_settlement_modifier
			multiplier = 100
			}
		}

	
	

	
	
	#Event Variables
	
	set_global_variable = {
		name = number_of_mega_earthquakes
		value = 0
	}

	set_global_variable = {
		name = number_of_mega_floods
		value = 0
	}
	
	set_global_variable = {
		name = number_of_mega_eruptions
		value = 0
	}
}
