ig_merchant_guilds = {
	color = hsv{ 0.06 0.60 0.89 }
	texture = "gfx/interface/icons/ig_icons/industrialists.dds"
	
	ideologies = {
		ideology_mercantile
	}
	
	enable = { 
		always = yes
		
		
	}
	
	traits = {
		ig_trait_great_works
		ig_trait_connected_markets
		ig_trait_withheld_advancements
	}	
	
	on_enable = {
		
		ig:ig_merchant_guilds = {
					set_ig_trait = ig_trait:ig_trait_great_works
					set_ig_trait = ig_trait:ig_trait_connected_markets
					set_ig_trait = ig_trait:ig_trait_withheld_advancements
				}

		}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
            
			
			OR = {
				is_pop_type = porters
				is_pop_type = merchants
				is_pop_type = artisans
				
            }   
    } 

	pop_weight = {
		value = 100
	
		
		add = {
			desc = "POP_PORTERS"
			if = {
				limit = { 
					is_pop_type = artisans
				}
				value = 300
			}
		}
		add = {
			desc = "POP_ARTISANS"
			if = {
				limit = { 
					is_pop_type = artisans
				}
				value = 100
			}
		}
		add = {
			desc = "POP_MERCHANTS"
			if = {
				limit = { 
					is_pop_type = artisans
				}
				value = 300
			}
		}

		
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}	
	}
	
	
	
	
	
	

	

	
	monarch_weight = {
		value = 0.5
	}	
	
	commander_weight = {
		value = 0.5
	}	
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
