ig_traditionalists = {
	color = hsv{ 0.50 0.30 0.67 }
	texture = "gfx/interface/icons/ig_icons/devout.dds"
	
	ideologies = {
		ideology_faith_and_culture
		ideology_classic_precedent
	}
	
	enable = {
		always = yes
	}
	
	traits = {
		ig_trait_religious_fervor
		ig_trait_uniting_sermons
		ig_trait_violated_traditions
	}		
	
	on_enable = {
		
		ig:ig_traditionalists = {
					set_ig_trait = ig_trait:ig_trait_religious_fervor
					set_ig_trait = ig_trait:ig_trait_uniting_sermons
					set_ig_trait = ig_trait:ig_trait_violated_traditions
				}

		}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = {
		OR = {
			is_pop_type = priests
			is_pop_type = chiefs
			is_pop_type = nobles
		}
	}	
	
	pop_weight = {
		value = 100
	
		add = {
			desc = "POP_CHIEFS"
			if = {
				limit = { 
					is_pop_type = chiefs
				}
				value = 400
			}
		}		
		
		
		add = {
			desc = "POP_PRIESTS"
			if = {
				limit = { 
					is_pop_type = priests
				}
				value = 400
			}
		}	
		
			
		
				
		
		
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}	

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}		
	}
	
	monarch_weight = {
		value = 0.25
	}
	
	commander_weight = {
		value = 1.0
	}	
	
	noble_chance = {
		value = 0.5
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.001

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
