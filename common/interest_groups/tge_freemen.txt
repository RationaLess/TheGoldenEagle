ig_freemen = {
	color = hsv{ 0.32 0.45 0.47 }
	texture = "gfx/interface/icons/ig_icons/rural_folk.dds"	
	
	ideologies = {
		ideology_sedentary_legacy
		ideology_mesoamerican_traditions
	}
	
	enable = { 
		always = yes
		
		
	}
	
	traits = {
		ig_trait_freedom_and_community
		ig_trait_peoples_approval
		ig_trait_local_resistance
	}	
	
	on_enable = {
		
		ig:ig_freemen = {
					set_ig_trait = ig_trait:ig_trait_freedom_and_community
					set_ig_trait = ig_trait:ig_trait_peoples_approval
					set_ig_trait = ig_trait:ig_trait_local_resistance
				}

		}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
            
		strata = poor
    } 

	pop_weight = {
		value = 100
	
		
		add = {
			desc = "POOR_STRATA"
			if = {
				limit = {
					strata = poor
				}
				value = 300
			}
		}
		
			
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}	
	}
	
	
	
	
	
	

	

	
	monarch_weight = {
		value = 0.5
	}	
	
	commander_weight = {
		value = 0.5
	}	
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
