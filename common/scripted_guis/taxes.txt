tge_character_wages_verylow = {
    scope = country
    effect = {
        set_variable = {
		name = character_wages
		value = 1
		}
		
		remove_modifier = verylow_character_wages
		remove_modifier = low_character_wages
		remove_modifier = medium_character_wages
		remove_modifier = high_character_wages
		remove_modifier = veryhigh_character_wages
		
		add_modifier = verylow_character_wages
		
    }
	
	is_shown = {
		NOT = { var:character_wages = 1 }
	}
}
showmodifier_verylow_character_wages = {
	effect = {
		add_modifier = {
			name = verylow_character_wages
		}
	}
}
