#Rations


mobilization_option_maize = {
	texture = "gfx/interface/icons/mobilization_options/basic_supplies.dds"

	possible = {
		market ?= {
			mg:maize ?= {
				market_goods_sell_orders > 0
			}
		}
	}
	
	unlocking_technologies = {
		army_reserves
	}	
	
	on_deactivate_while_mobilized = {
		custom_tooltip = {
			text = mobilization_option_it_hurts_morale_when_you_remove_supplies_while_in_combat_tt
			every_combat_unit = {
				add_morale = {
					value = morale
					multiply = -0.5
				}
			}
		}
	}
	
	upkeep_modifier = {
		goods_input_maize_add = 3
	}
	
	unit_modifier = {
		unit_morale_loss_mult = -0.1
		unit_offense_mult = 0.1
		unit_defense_mult = 0.1
	}
	ai_weight = {
		value = 1
	}

	group = supplies
}
