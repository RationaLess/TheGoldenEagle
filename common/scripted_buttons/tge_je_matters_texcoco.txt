matters_of_texcoco_proclaim_tepanecapan = {
	name = "matters_of_texcoco_proclaim_tepanecapan_button"
	desc = "matters_of_texcoco_proclaim_tepanecapan_button_desc"

	visible = { always = yes }

	possible = {
		NOT = {
			exists = c:TEPA
			this = c:TEPA
			}
		
		at_war = no
		country_has_primary_culture = cu:tepanec
		
		any_subject_or_below = {
			is_subject_type = subject_type_vassal
			OR = {
				country_has_primary_culture = cu:tepanec
			}
			count >= 1
		}
		
	}

	effect = {
		change_tag = TEPA
		custom_tooltip = {
				text = effect_texcoco_struggle_advance_10
				effect_texcoco_struggle_advance_10
				}
		custom_tooltip = {
				text = matters_of_texcoco_proclaim_tepanecapan_subject
					every_subject_or_below = {
					limit = {
						country_has_primary_culture = cu:tepanec
						}
					root = {
						annex_with_incorporation = prev
						}
				}
			}
		
	}

	ai_chance = {
		base = 100
	}
}
