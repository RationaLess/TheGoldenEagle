confederation_establish_council_button = {
	name = "confederation_establish_council_button"
	desc = "confederation_establish_council_button_desc"

	visible = { always = yes }

	possible = {
		
		OR = {
			has_law = law_type:law_dedicated_government
			has_law = law_type:law_bureaucratic_government
		}
	
		legitimacy > 75
		ig:ig_ancillaries = {
			is_in_government = yes
		}
		
		has_law = law_type:law_alliance
		NOT = {
			any_subject_or_below = {
				is_subject_type = subject_type_confederation_candidate 
			}
		}
		
		custom_tooltip = {
				text = je_matters_of_texcoco_confederate
				AND = {
					has_variable = confederations_num
					var:confederations_num >= 1
					NOT = {
						has_law = law_type:law_bellicose
						}
					}
				}
	}

	effect = {
		activate_law = law_type:law_council
		trigger_event = { id = forging_alliances.2 days = 0 popup = yes }
		
		
	}

	ai_chance = {
		base = 100
	}
}

entrenched_dynasts_negotiation_button = {
	name = "entrenched_dynasts_negotiation_button"
	desc = "entrenched_dynasts_negotiation_button_desc"

	visible = { always = yes }

	possible = {
		
		loyalist_fraction = {
			value > 0.25
			strata = rich
			}
		ig:ig_aristocracy = {
					ig_approval >= 10
			OR = {
				is_being_suppressed = yes 
				ig_clout < 0.4
				}
			}
	}

	effect = {
		custom_tooltip = {
				text = je_entrenched_dynasts_negotiation_effect
				set_variable = entrenched_dynasts_negotiation
			}
		
	}

	ai_chance = {
		base = 100
	}
}
