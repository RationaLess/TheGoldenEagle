ideology_moderate = {
	icon = "gfx/interface/icons/ideology_icons/ideology_leader/ideology_leader_no_ideology.dds"
	
	show_in_list = no
	
	character_ideology = yes
	
	possible = {
		always = yes
		
		scope:interest_group = {
			NOR = {
				has_variable = communist_ig_var
				has_variable = chose_vanguardism
				has_variable = chose_fascism
				has_variable = chose_conservatism
				has_variable = chose_ethno_nationalism
			}
		}
	}
	
	leader_weight = {
		value = 100	
		# angry IG's more likely to have opinions
		if = {
			limit = {
				scope:interest_group = {
					ig_approval < -5
				}
			}
			add = -75
		}
		# what makes a man turn neutral? lust for gold? power? or were you just born with a heart full of neutrality?
		if = {
			limit = {
				scope:interest_group = {
					ig_approval = 0
				}
			}
			add = 1
		}
	}
}

		}
	lawgroup_land_organization = {

	lawgroup_military_reforms = {
			law_no_standing_army = strongly_disapprove
			law_warrior_guilds = neutral
			law_military_education = approve 
			law_honorary_titles = strongly_approve
		}
	
	
	
	possible = {
		scope:interest_group = {
			OR = {
				is_interest_group_type = ig_aristocracy
				is_interest_group_type = ig_ancillaries
				is_interest_group_type = ig_warrior_guilds
				is_interest_group_type = ig_pochteca
				is_interest_group_type = ig_scholars
			}
		}
		
		}
	
	leader_weight = {
		value = 25
		
		#Warrior guilds REALLY like this stuff
		if = {
			limit = {
			scope:interest_group = {
					is_interest_group_type = ig_warrior_guilds
				}
			}
		add = 175
		}
		#Aristocrats are more likely to be into building an empire
		if = {
			limit = {
			scope:interest_group = {
					is_interest_group_type = ig_aristocracy
				}
			}
		add = 50
		}
		#Powerful Warrior Guilds help the Imperial cause
		if = {
			limit = {
				owner = {
					exists = ig:ig_warrior_guilds
					ig:ig_warrior_guilds = {
						is_powerful = yes
					}
				}
			}
			add = 50
		}
		# Bellicose foreign policy makes it more likely
		if = {
			limit = {
				owner = {
					has_law = law_type:law_bellicose
				}
			}
			add = 50
		}
		#Only IGs which are happy will go imperialist
		if = {
			limit = {
				scope:interest_group = {
					ig_approval > 5
				}
			}
			add = 50
		}
		#Only IGs which are happy will go imperialist
		if = {
			limit = {
				scope:interest_group = {
					ig_approval > 5
				}
			}
			add = 50
		}
		#Boosted by Imperial Fervor
		if = {
			limit = {
				owner = {
				has_journal_entry = je_imperial_fervor
				}
			}
			add = 50
		}
		# More likely for GP's
		if = {
			limit = {
				owner = {
					country_rank = rank_value:great_power
				}
			}
			add = 25
		}
	
	}

}

	possible = {
		owner = {

			OR = {
				has_law = law_type:law_city_state
				has_law = law_type:law_alliance
				}
			NOT = {
				OR = {
					has_law = law_type:law_dedicated_government
					has_law = law_type:law_bureaucratic_government
					}
				}
		}
		scope:interest_group = {
			OR = {
				is_interest_group_type = ig_aristocracy
				is_interest_group_type = ig_ancillaries
				is_interest_group_type = ig_pochteca
				is_interest_group_type = ig_freemen
				is_interest_group_type = ig_scholars
			}
		}
		
		}
	
	leader_weight = {
		value = 25
		

		
		#Ancillaries should aim to establish confederations
		if = {
			limit = {
			scope:interest_group = {
					is_interest_group_type = ig_ancillaries
				}
			}
		add = 175
		}
		# Low prestige counties are likely to go for confederations
		if = {
			limit = {
				owner = {
					prestige <= 100
				}
			}
			add = 100
		}
		# REALLY low prestige counties are more likely to go for confederations
		if = {

				}
			}
			multiply = 0.1
		}
	
	}
