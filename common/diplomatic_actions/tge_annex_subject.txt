integrate_subject = {
	requires_approval = no
	should_notify_third_parties = no
	show_effect_in_tooltip = yes

	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		scope:target_country = {
			is_subject_of = prev
		}
	}
	
	possible = {	
		NOT = { has_war_with = scope:target_country }	
		infamy < 25
		is_at_war = no
		custom_tooltip = {
			text = empire_sprawl_trigger_positive
			empire_sprawl_value > 0
			}
		scope:target_country = {
			is_subject_of = prev
			NOT = {
				has_truce_with = prev
				}
			}
	}
	
	accept_effect = {
		add_modifier = {
			name = integrating_subject
			months = 120
			multiplier = {
				value = scope:target_country.total_population 
				divide = 100000
				}
			is_decaying = yes
			}
	
		annex_with_incorporation = scope:target_country
	}
	
	ai = {
		propose_score = {
			value = 100
		}
	
		will_propose = {
			
			set_variable = {
				name = diplo_comparison
				value = {
					value = scope:target_country.total_population 
					divide = 100000
					}
				}
			
			bureaucracy >= diplo_comparison
			
			
		}
	}
}
