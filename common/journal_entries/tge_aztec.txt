		}

		trigger_event = matters_of_texcoco.1
	}

	possible = {
		is_subject = no
		is_at_war = no
	}
	
	fail = {
		is_subject = yes
	}
	
	on_monthly_pulse = {
		random_events = {
			50 = 0
			10 = matters_of_texcoco.2
			5 = matters_of_texcoco.3
		}
	}

	modifiers_while_active = {
	}
	
	complete = {
		OR = {
			AND = {
				country_or_subject_owns_entire_state_region = STATE_TENOCHTITLAN

				country_or_subject_owns_entire_state_region = STATE_CHALCO
			}
			c:TLX = {
				has_diplomatic_pact = {
					who = root
					type = alliance
				}
			}
			custom_tooltip = {
				text = je_matters_of_texcoco_confederate
				AND = {
					has_variable = confederations_num
					var:confederations_num >= 1

						has_law = law_type:law_bellicose
