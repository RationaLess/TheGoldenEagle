je_imperial_fervor = {

	should_be_pinned_by_default = yes
	weight = 200

modifiers_while_active = {
		imperial_potential
	}

status_desc = {
		first_valid = {
			triggered_desc = { 
				desc = je_imperial_fervor_goal
			}
		}
	}

is_shown_when_inactive = {
		
		always = yes
		
	}

possible = {
		prestige > 100
		is_subject = no
		is_at_war = no
		has_law = law_type:law_bellicose
		
		
		any_interest_group = {
				is_powerful = yes
				is_in_government = yes
				OR = {
					is_interest_group_type = ig_warrior_guilds
					leader = {
						OR = {
							has_ideology = ideology:ideology_leader_imperialist
						}
					}
				}
			}
		
	}

immediate = {

	if = {
			limit = {
				NOT = {
					has_variable = imperial_fervor
				}
			}
			set_variable = {
				name = imperial_fervor
				value = 50
			}
		}

	trigger_event = {
			id = imperial.1
			days = 0
			popup = yes
			}
	}


progressbar = yes 


current_value = {
		value = 0
		if = {
			limit = {
				has_variable = imperial_fervor
				has_journal_entry = je_imperial_fervor
			}
			value = var:imperial_fervor
		}
	}
	
goal_add_value = {
		value = 100
	}

on_monthly_pulse = {
	change_variable = {
				name = imperial_fervor
				add = {
					value = -1
					multiply = {
						value = 1
							limit = {
								if = { has_law = law_type:law_law_origin_myth }
								multiply = 0.5
							}
							limit = {
								if = { has_law = law_type:law_imperial_land }
								multiply = 0.5
							}
							limit = {
								if = { has_law = law_type:law_honorary_titles }
								multiply = 0.5
							}
						}
				}
			}
	}
	

fail = {
		custom_tooltip = {
			text = je_imperial_fervor_fail
			var:imperial_fervor <= 0
		}
	}

}
