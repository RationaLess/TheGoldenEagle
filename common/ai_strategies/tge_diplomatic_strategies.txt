
ai_strategy_royal_affairs = {
	icon = "gfx/interface/icons/ai_strategy_icons/territorial_expansion.dds"
	
	type = diplomatic
	
	undesirable_infamy_level = {
		value = 50
	}
	
	unacceptable_infamy_level = {
		value = 100
	}
	
	ideological_opinion_effect_mult = {
		value = 0.75
	}
	
	wargoal_maneuvers_fraction = {
		value = 0.15
	}		
	
	diplomatic_play_neutrality = {
		value = 25
	}
	
	diplomatic_play_boldness = {
		value = 25
	}
	
	recklessness = {
		value = 0.25
	}
	
	aggression = {
		value = 1
		

		# Less aggressive against countries we're just antagonistic against, rather than belligerent/domineering
		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}					
			}
			multiply = 0.25
		}	

		# More aggressive against ideological enemies, a bit less aggressive against ideological allies
		if = {
			limit = {
				"ai_ideological_opinion(scope:target_country)" < 0
			}
			
			multiply = {
				value = "ai_ideological_opinion(scope:target_country)"
				multiply = -0.2
				add = 1
				max = 4.0
			}				
		}	
		else = {
			multiply = {
				value = "ai_ideological_opinion(scope:target_country)"
				multiply = -0.05
				add = 1
				min = 0.5
			}			
		}			
	}

	building_group_weights = {
		bg_army = 1.5
	}
	
	secret_goal_weights = {
		protect = 0.5
		conquer = 1.5
	}

	wargoal_weights = {
		protectorate = 5.0
		conquer_state = 1.5
		return_state = 1.5
	}	
		
	possible = {
		always = yes
	}

	weight = {
		value = 0

	}
}

ai_strategy_form_confederation = {
	icon = "gfx/interface/icons/ai_strategy_icons/unused/protect_region.dds"
	
	type = diplomatic
	
	undesirable_infamy_level = {
		value = 0
	}
	
	unacceptable_infamy_level = {
		value = 50
	}
	
	ideological_opinion_effect_mult = {
		value = -0.5
	}
	
	wargoal_maneuvers_fraction = {
		value = -0.1
	}	
	
	diplomatic_play_neutrality = {
		value = 0
	}
	
	diplomatic_play_boldness = {
		value = -25
	}
	
	recklessness = {
		value = -0.5
	}
	
	aggression = {
		value = -0.75

		if = {
			limit = { 
				NOT = { is_country_type = unrecognized }
				scope:target_country = { is_country_type = unrecognized } 
			}
			add = 0.25
			
			if = {
				limit = { has_technology_researched = civilizing_mission }
				add = 0.5
			}			
		}

		# Less aggressive against countries we're just antagonistic against, rather than belligerent/domineering
		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}					
			}
			multiply = 0.1
		}

		# More aggressive against ideological enemies, a bit less aggressive against ideological allies
		if = {
			limit = {
				"ai_ideological_opinion(scope:target_country)" < 0
			}
			
			multiply = {
				value = "ai_ideological_opinion(scope:target_country)"
				multiply = -0.2
				add = 1
				max = 4.0
			}				
		}	
		else = {
			multiply = {
				value = "ai_ideological_opinion(scope:target_country)"
				multiply = -0.05
				add = 1
				min = 0.5
			}			
		}		
	}
	 
	building_group_weights = {
		bg_technology = 1.25
	}
	
	secret_goal_weights = {
		protect = 1.5
		dominate = 1.5
		reconcile = 1.5
	}
	
	secret_goal_scores = {
	
		befriend = {
			value = 300
			if = {
				limit = {	
					scope:target_country = {
						is_adjacent_to_country = root
					}
				}
			
			add = 900
			}
		}

	}

	possible = {
		has_technology_researched = confederation
		has_journal_entry = je_forging_alliances
		
		OR = {
			has_law = law_type:law_alliance
			has_law = law_type:law_city_state
		}
		NOT = {
			has_law = law_type:law_bellicose
		}

	}
	
	weight = {
		value = 50
		
		if = {
			limit = { country_rank < rank_value:major_power }
			add = 100
		}
	
		IF = {
			limit = { NOT = {
				has_journal_entry = je_forging_alliances
				} }
			multiply = 0	
			}
	
	}
}

	}
	
	recklessness = {
		value = 4
	}
	
	aggression = {
		value = 2
		
		if = {
			limit = { 
				NOT = { is_country_type = unrecognized }
				scope:target_country = { is_country_type = unrecognized } 
			}
			add = 1
		}	

		# Less aggressive against countries we're just antagonistic against, rather than belligerent/domineering
		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}

			multiply = 0.25
		}	

		# More aggressive against ideological enemies, a bit less aggressive against ideological allies
		if = {
			limit = {
				"ai_ideological_opinion(scope:target_country)" < 0
			}
			
			multiply = {
				value = "ai_ideological_opinion(scope:target_country)"
				multiply = -0.2
				add = 1
				max = 4.0
			}				
		}	
		else = {
			multiply = {
				value = "ai_ideological_opinion(scope:target_country)"
				multiply = -0.05
				add = 1
				min = 0.5
			}			
		}			
	}

	building_group_weights = {
		bg_army = 1.5
	}
	
	secret_goal_weights = {
		dominate = 4
		conquer = 4
	}
	
	secret_goal_scores = {
		#imperialist AIs should be psychotic
		dominate = {
			if = {
				limit = {
					can_reach_target_country = yes
					has_strategic_adjacency = scope:target_country
					scope:target_country.country_rank <= root.country_rank
				}
				add = 1000
			}
		}
		conquer = {
			if = {
				limit = {
					can_reach_target_country = yes
					has_strategic_adjacency = scope:target_country
					scope:target_country.country_rank <= root.country_rank
				}
			add = 1000
		}
	}
	
	
	wargoal_weights = {
		conquer_state = 4
		protectorate = 8
		return_state = 4
	}	
		
	
	possible = {
		
		OR = {
			ig:ig_warrior_guilds = { is_powerful = yes }
			has_law = law_type:law_bellicose
			has_law = law_type:law_tributary_empire
		}
		NOT = {
			has_law = law_type:law_appeasing
		}

	}
	
	weight = {
		value = 25
			
		if = {
			limit = { has_law = law_type:law_bellicose }
			multiply = 2
		}	
		if = {
			limit = { 
			OR = {
				has_law = law_type:law_tributary_empire
				has_law = law_type:law_territory_integration
				}
			multiply = 2
			}				
		}
		if = {
			limit = { 
			OR = {
				has_law = law_type:law_council
				has_law = law_type:law_monarchy
				}
			multiply = 2
			}				
		}
		IF = {
			limit = {
				has_journal_entry = je_kindling_the_empire
				}
			multiply = 4	
			}
	}
}
